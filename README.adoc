link:rhdm72-ephemeral.yaml[rhdm72-ephemeral.yaml]

[source, text]
.*1. 创建 ImageStream*
----
$ oc create -f rhdm72-image-streams.yaml -n openshift
----

[source, text]
.*2. 创建模版*
----
$ oc create -f rhdm72-ephemeral.yaml 
----

.*3. 部署*
